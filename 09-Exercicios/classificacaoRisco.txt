import javax.swing.JOptionPane;
public class ClassificacaoRisco {

    public static void main(String[] args) {

        // Entrada de dados
        String nome = JOptionPane.showInputDialog("Digite o nome do cliente:");
        double salario = Double.parseDouble(JOptionPane.showInputDialog("Digite o salário mensal:"));
        int score = Integer.parseInt(JOptionPane.showInputDialog("Digite o score de crédito:"));

        // Classificação
        String classificacao;

        if (score > 800) {
            classificacao = "Risco Baixo";
        } else if (score >= 600) {
            classificacao = "Risco Médio";
        } else {
            classificacao = "Risco Alto";
        }

        // Saída
        JOptionPane.showMessageDialog(
                null,
                "===== CLASSIFICAÇÃO DE RISCO =====\n"
                        + "Cliente: " + nome + "\n"
                        + "Salário: R$ " + salario + "\n"
                        + "Score: " + score + "\n"
                        + "Classificação: " + classificacao
        );
    }
}
