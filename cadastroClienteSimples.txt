import javax.swing.JOptionPane;

public class CadastroCliente {
    public static void main(String[] args) {

        // Entrada de dados
        String nome = JOptionPane.showInputDialog("Digite o nome do cliente:");

        int idade = Integer.parseInt(JOptionPane.showInputDialog("Digite a idade:"));

        double salario = Double.parseDouble(JOptionPane.showInputDialog("Digite o salário mensal:") );
        
        double emprestimo = Double.parseDouble(JOptionPane.showInputDialog
        ("Digite o valor do empréstimo desejado:"));

        // Saída de dados
        String mensagem = "===== DADOS DO CLIENTE =====\n"
                + "Nome: " + nome + "\n"
                + "Idade: " + idade + "\n"
                + "Salário: R$ " + salario + "\n"
                + "Empréstimo desejado: R$ " + emprestimo;

        JOptionPane.showMessageDialog(null, mensagem);
    }
}
