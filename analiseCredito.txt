import javax.swing.JOptionPane;

public class AnaliseCreditoSimples {

    public static void main(String[] args) {

        // Entrada de dados
        String nome = JOptionPane.showInputDialog("Digite o nome do cliente:");

        double salario = Double.parseDouble(
                JOptionPane.showInputDialog("Digite o salário mensal:")
        );

        double parcela = Double.parseDouble(
                JOptionPane.showInputDialog("Digite o valor da parcela desejada:")
        );

        // Cálculo
        double limite = salario * 0.30;

        String resultado;

        if (parcela <= limite) {
            resultado = "Crédito pode ser analisado.";
        } else {
            resultado = "Crédito negado.";
        }

        // Confirmação simples
        int confirmacao = JOptionPane.showConfirmDialog(
                null,
                "Deseja ver o resultado da análise?",
                "Confirmação",
                JOptionPane.YES_NO_OPTION
        );

        if (confirmacao == JOptionPane.YES_OPTION) {

            // Opções extras
            String[] opcoes = {"Mostrar Resumo", "Mostrar Apenas Resultado"};

            int escolha = JOptionPane.showOptionDialog(
                    null,
                    "Escolha como deseja visualizar:",
                    "Opções",
                    JOptionPane.DEFAULT_OPTION,
                    JOptionPane.INFORMATION_MESSAGE,
                    null,
                    opcoes,
                    opcoes[0]
            );

            if (escolha == 0) {
                // Mostra tudo
                JOptionPane.showMessageDialog(
                        null,
                        "===== ANÁLISE DE CRÉDITO =====\n"
                                + "Cliente: " + nome + "\n"
                                + "Salário: R$ " + salario + "\n"
                                + "Parcela: R$ " + parcela + "\n"
                                + "Limite (30%): R$ " + limite + "\n"
                                + "Resultado: " + resultado
                );
            } else {
                // Mostra só resultado
                JOptionPane.showMessageDialog(
                        null,
                        "Resultado: " + resultado
                );
            }
        } else {
            JOptionPane.showMessageDialog(null, "Operação cancelada.");
        }
    }
}
